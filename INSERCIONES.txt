CREATE TABLE EMPLEADO (
    ID_EMP NUMBER(4),
    RFC CHAR(13),
    NSS CHAR(11),
    FECHANAC DATE,
    GENERO VARCHAR2(9),
    PILAEMP VARCHAR2(20),
    PATEMP VARCHAR2(15),
    MATEMP VARCHAR2(15),
    CALLE_EMP VARCHAR2(15),
    COLEMP VARCHAR2(15),
    ESTADOEMP VARCHAR2(20),
    CONSTRAINT PK_EMP PRIMARY KEY (ID_EMP),
    CONSTRAINT RFC_UQ UNIQUE (RFC),
    CONSTRAINT NSS_UQ UNIQUE (NSS),
    CONSTRAINT GENERO_CK CHECK (GENERO IN ('MASCULINO', 'FEMENINO'))
